# –ö–æ–Ω—Å–ø–µ–∫—Ç –ø–æ —Ä–µ–≥—É–ª—è—Ä–Ω–∏–º –≤–∏—Ä–∞–∑–∞–º (regex) —É Python

–†–µ–≥—É–ª—è—Ä–Ω—ñ –≤–∏—Ä–∞–∑–∏ (regex) ‚Äî —Ü–µ –ø–æ—Ç—É–∂–Ω–∏–π —ñ–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç –¥–ª—è –ø–æ—à—É–∫—É —ñ –º–∞–Ω—ñ–ø—É–ª—è—Ü—ñ—ó –∑ —Ç–µ–∫—Å—Ç–æ–º. –£ Python —ó—Ö –º–æ–∂–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞—Ç–∏ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é –º–æ–¥—É–ª—è re. –í–æ–Ω–∏ –¥–æ–∑–≤–æ–ª—è—é—Ç—å –≤–∏–∫–æ–Ω—É–≤–∞—Ç–∏ –ø–æ—à—É–∫ –∑–∞ —à–∞–±–ª–æ–Ω–æ–º —É —Ä—è–¥–∫–∞—Ö.

## –û—Å–Ω–æ–≤–Ω—ñ —Ñ—É–Ω–∫—Ü—ñ—ó –º–æ–¥—É–ª—è re:
- re.search(pattern, string)

–®—É–∫–∞—î –ø–µ—Ä—à–µ –≤—Ö–æ–¥–∂–µ–Ω–Ω—è —à–∞–±–ª–æ–Ω—É –≤ —Ä—è–¥–∫—É —ñ –ø–æ–≤–µ—Ä—Ç–∞—î –æ–±'—î–∫—Ç –º–∞—Ç—á—É –∞–±–æ None, —è–∫—â–æ –∑–±—ñ–≥ –Ω–µ –∑–Ω–∞–π–¥–µ–Ω–æ.

```python
import re
result = re.search(r'\d+', 'Hello 123 World')  # –®—É–∫–∞—î –ø–µ—Ä—à–µ —á–∏—Å–ª–æ
print(result.group())  # –í–∏–≤–µ–¥–µ: 123
```

- re.match(pattern, string)

–ü–µ—Ä–µ–≤—ñ—Ä—è—î, —á–∏ –ø–æ—á–∏–Ω–∞—î—Ç—å—Å—è —Ä—è–¥–æ–∫ —ñ–∑ –∑–∞–∑–Ω–∞—á–µ–Ω–æ–≥–æ —à–∞–±–ª–æ–Ω—É.

```python
result = re.match(r'Hello', 'Hello World')  # –ü–æ–≤–µ—Ä—Ç–∞—î –∑–±—ñ–≥, —è–∫—â–æ —Ä—è–¥–æ–∫ –ø–æ—á–∏–Ω–∞—î—Ç—å—Å—è –∑ 'Hello'
```

- re.findall(pattern, string)

–ü–æ–≤–µ—Ä—Ç–∞—î —Å–ø–∏—Å–æ–∫ —É—Å—ñ—Ö –≤—Ö–æ–¥–∂–µ–Ω—å —à–∞–±–ª–æ–Ω—É –≤ —Ä—è–¥–∫—É.

```python
result = re.findall(r'\d+', 'Hello 123 and 456')  # –®—É–∫–∞—î –≤—Å—ñ —á–∏—Å–ª–∞
print(result)  # –í–∏–≤–µ–¥–µ: ['123', '456']
```

- re.sub(pattern, replacement, string)

–ó–∞–º—ñ–Ω—é—î –≤—Å—ñ –≤—Ö–æ–¥–∂–µ–Ω–Ω—è —à–∞–±–ª–æ–Ω—É –Ω–∞ —ñ–Ω—à–∏–π —Ä—è–¥–æ–∫.

```python
result = re.sub(r'\d+', 'NUMBER', 'Order 123 and 456')  # –ó–∞–º—ñ–Ω—é—î –≤—Å—ñ —á–∏—Å–ª–∞
print(result)  # –í–∏–≤–µ–¥–µ: Order NUMBER and NUMBER
```

- re.split(pattern, string)

–†–æ–∑–±–∏–≤–∞—î —Ä—è–¥–æ–∫ –∑–∞ —à–∞–±–ª–æ–Ω–æ–º.

```python
result = re.split(r'\s+', 'Hello   World')  # –†–æ–∑–±–∏–≤–∞—î —Ä—è–¥–æ–∫ –∑–∞ –ø—Ä–æ–±—ñ–ª–∞–º–∏
print(result)  # –í–∏–≤–µ–¥–µ: ['Hello', 'World']
```

## –°–ø–µ—Ü—ñ–∞–ª—å–Ω—ñ —Å–∏–º–≤–æ–ª–∏ –≤ —Ä–µ–≥—É–ª—è—Ä–Ω–∏—Ö –≤–∏—Ä–∞–∑–∞—Ö:
`.` ‚Äî –±—É–¥—å-—è–∫–∏–π —Å–∏–º–≤–æ–ª, –æ–∫—Ä—ñ–º —Å–∏–º–≤–æ–ª—É –Ω–æ–≤–æ–≥–æ —Ä—è–¥–∫–∞.

```python
re.search(r'H.llo', 'Hello')  # –ü–æ–≤–µ—Ä–Ω–µ –∑–±—ñ–≥
```

`^` ‚Äî –ø–æ—á–∞—Ç–æ–∫ —Ä—è–¥–∫–∞.

```python
re.search(r'^Hello', 'Hello World')  # –ü–æ–≤–µ—Ä–Ω–µ –∑–±—ñ–≥
```

`$` ‚Äî –∫—ñ–Ω–µ—Ü—å —Ä—è–¥–∫–∞.

```python
re.search(r'World$', 'Hello World')  # –ü–æ–≤–µ—Ä–Ω–µ –∑–±—ñ–≥
```

`\d` ‚Äî –±—É–¥—å-—è–∫–∞ —Ü–∏—Ñ—Ä–∞.

```python
re.findall(r'\d+', '123 apples')  # –ó–Ω–∞–π–¥–µ: ['123']
```

`\w` ‚Äî –±—É–¥—å-—è–∫–∏–π –±—É–∫–≤–µ–Ω–∏–π —Å–∏–º–≤–æ–ª –∞–±–æ —Ü–∏—Ñ—Ä–∞ (–∞–ª—Ñ–∞–≤—ñ—Ç + —Ü–∏—Ñ—Ä–∏).

```python
re.findall(r'\w+', 'Hello 123!')  # –ó–Ω–∞–π–¥–µ: ['Hello', '123']
```

`\s` ‚Äî –±—É–¥—å-—è–∫–∏–π –ø—Ä–æ–±—ñ–ª—å–Ω–∏–π —Å–∏–º–≤–æ–ª (–ø—Ä–æ–±—ñ–ª, —Ç–∞–±—É–ª—è—Ü—ñ—è, –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫).

```python
re.split(r'\s+', 'Hello   World')  # –†–æ–∑—ñ–±'—î –Ω–∞ ['Hello', 'World']
```

## –ö–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä–∏:
–ö–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä–∏ –≤ —Ä–µ–≥—É–ª—è—Ä–Ω–∏—Ö –≤–∏—Ä–∞–∑–∞—Ö (regex) –≤–∏–∑–Ω–∞—á–∞—é—Ç—å, —Å–∫—ñ–ª—å–∫–∏ —Ä–∞–∑—ñ–≤ –ø–µ–≤–Ω–∏–π –µ–ª–µ–º–µ–Ω—Ç (—Å–∏–º–≤–æ–ª –∞–±–æ –≥—Ä—É–ø–∞ —Å–∏–º–≤–æ–ª—ñ–≤) –º–∞—î –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è –≤ —Ä—è–¥–∫—É, —â–æ–± –≤–≤–∞–∂–∞—Ç–∏—Å—è –∑–±—ñ–≥–æ–º. –î–∞–≤–∞–π—Ç–µ —Ä–æ–∑–±–µ—Ä–µ–º–æ –∫–æ–∂–µ–Ω –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –¥–æ–∫–ª–∞–¥–Ω—ñ—à–µ –∑ –ø—Ä–∏–∫–ª–∞–¥–∞–º–∏, —â–æ–± –±—É–ª–æ –∑—Ä–æ–∑—É–º—ñ–ª–æ, —è–∫ –≤–æ–Ω–∏ –ø—Ä–∞—Ü—é—é—Ç—å.

`*` ‚Äî 0 –∞–±–æ –±—ñ–ª—å—à–µ –ø–æ–≤—Ç–æ—Ä–µ–Ω—å

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –≤–∫–∞–∑—É—î, —â–æ —Å–∏–º–≤–æ–ª –º–æ–∂–µ –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è –±—É–¥—å-—è–∫—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Ä–∞–∑—ñ–≤, –≤–∫–ª—é—á–∞—é—á–∏ 0 —Ä–∞–∑—ñ–≤.

–ü—Ä–∏–∫–ª–∞–¥:
```python
import re

# –®—É–∫–∞—î 'ab', –ø—ñ—Å–ª—è —è–∫–æ–≥–æ –π–¥–µ –±—É–¥—å-—è–∫–∞ –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–∏–º–≤–æ–ª—ñ–≤ 'c' (0 –∞–±–æ –±—ñ–ª—å—à–µ)
result = re.search(r'abc*', 'ab')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'ab', –±–æ 'c' –º–æ–∂–µ –±—É—Ç–∏ 0 —Ä–∞–∑—ñ–≤

result = re.search(r'abc*', 'abcc')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'abcc', –±–æ 'c' –∑—É—Å—Ç—Ä—ñ—á–∞—î—Ç—å—Å—è 2 —Ä–∞–∑–∏
```

abc* –æ–∑–Ω–∞—á–∞—î, —â–æ –ø—ñ—Å–ª—è ab –º–æ–∂–µ –π—Ç–∏ 0 –∞–±–æ –±—ñ–ª—å—à–µ —Å–∏–º–≤–æ–ª—ñ–≤ c. –£ –ø–µ—Ä—à–æ–º—É –≤–∏–ø–∞–¥–∫—É –∑–±—ñ–≥–∞—î—Ç—å—Å—è –ø—Ä–æ—Å—Ç–æ ab, –∞ –≤ –¥—Ä—É–≥–æ–º—É ‚Äî abcc.

`+` ‚Äî 1 –∞–±–æ –±—ñ–ª—å—à–µ –ø–æ–≤—Ç–æ—Ä–µ–Ω—å

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –≤–∏–º–∞–≥–∞—î, —â–æ–± —Å–∏–º–≤–æ–ª –∑—É—Å—Ç—Ä—ñ—á–∞–≤—Å—è –ø—Ä–∏–Ω–∞–π–º–Ω—ñ –æ–¥–∏–Ω —Ä–∞–∑, –∞–ª–µ –º–æ–∂–µ –±—É—Ç–∏ —ñ –±—ñ–ª—å—à–µ.

–ü—Ä–∏–∫–ª–∞–¥:
```python
import re

# –®—É–∫–∞—î 'ab', –ø—ñ—Å–ª—è —è–∫–æ–≥–æ –π–¥–µ —Ö–æ—á–∞ –± –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª 'c'
result = re.search(r'abc+', 'ab')
print(result)  # –í–∏–≤–µ–¥–µ: None, –±–æ 'c' –Ω–µ –∑—É—Å—Ç—Ä—ñ—á–∞—î—Ç—å—Å—è

result = re.search(r'abc+', 'abc')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'abc', –±–æ 'c' –∑—É—Å—Ç—Ä—ñ—á–∞—î—Ç—å—Å—è 1 —Ä–∞–∑

result = re.search(r'abc+', 'abcccc')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'abcccc', –±–æ 'c' –∑—É—Å—Ç—Ä—ñ—á–∞—î—Ç—å—Å—è –±—ñ–ª—å—à–µ 1 —Ä–∞–∑
```

abc+ –æ–∑–Ω–∞—á–∞—î, —â–æ –ø—ñ—Å–ª—è ab –ø–æ–≤–∏–Ω–µ–Ω –±—É—Ç–∏ —Ö–æ—á–∞ –± –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª c.

`?` ‚Äî 0 –∞–±–æ 1 –ø–æ–≤—Ç–æ—Ä–µ–Ω–Ω—è

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –æ–∑–Ω–∞—á–∞—î, —â–æ —Å–∏–º–≤–æ–ª –º–æ–∂–µ –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è –∞–±–æ –æ–¥–∏–Ω —Ä–∞–∑, –∞–±–æ –≤–∑–∞–≥–∞–ª—ñ –Ω–µ –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è.

–ü—Ä–∏–∫–ª–∞–¥:
```python
import re

# –®—É–∫–∞—î 'ab', –ø—ñ—Å–ª—è —è–∫–æ–≥–æ –º–æ–∂–µ –±—É—Ç–∏ –∞–±–æ 1, –∞–±–æ 0 —Å–∏–º–≤–æ–ª—ñ–≤ 'c'
result = re.search(r'abc?', 'ab')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'ab', –±–æ 'c' –º–æ–∂–µ –±—É—Ç–∏ 0 —Ä–∞–∑—ñ–≤

result = re.search(r'abc?', 'abc')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'abc', –±–æ 'c' –∑—É—Å—Ç—Ä—ñ—á–∞—î—Ç—å—Å—è 1 —Ä–∞–∑
```

abc? –æ–∑–Ω–∞—á–∞—î, —â–æ –ø—ñ—Å–ª—è ab –º–æ–∂–µ –±—É—Ç–∏ 0 –∞–±–æ 1 —Å–∏–º–≤–æ–ª c.

`{n}` ‚Äî —Ä—ñ–≤–Ω–æ n –ø–æ–≤—Ç–æ—Ä–µ–Ω—å

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –∑–∞–¥–∞—î —Ç–æ—á–Ω—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å –ø–æ–≤—Ç–æ—Ä–µ–Ω—å —Å–∏–º–≤–æ–ª—É.

–ü—Ä–∏–∫–ª–∞–¥:
```python
import re

# –®—É–∫–∞—î —Ä—ñ–≤–Ω–æ —Ç—Ä–∏ —Å–∏–º–≤–æ–ª–∏ 'a'
result = re.search(r'a{3}', 'aa')
print(result)  # –í–∏–≤–µ–¥–µ: None, –±–æ —î —Ç—ñ–ª—å–∫–∏ 2 'a'

result = re.search(r'a{3}', 'aaa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aaa', –±–æ —î —Ä—ñ–≤–Ω–æ 3 'a'

result = re.search(r'a{3}', 'aaaa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aaa', –±–æ –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä —à—É–∫–∞—î —Ä—ñ–≤–Ω–æ 3 —Å–∏–º–≤–æ–ª–∏ 'a'
```

a{3} —à—É–∫–∞—î —Ä—ñ–≤–Ω–æ 3 —Å–∏–º–≤–æ–ª–∏ a. –ù–∞–≤—ñ—Ç—å —è–∫—â–æ –≤ —Ä—è–¥–∫—É –±—ñ–ª—å—à–µ —Å–∏–º–≤–æ–ª—ñ–≤, –±—É–¥–µ –∑–Ω–∞–π–¥–µ–Ω–æ —Ç—ñ–ª—å–∫–∏ 3.

`{n,}` ‚Äî –Ω–µ –º–µ–Ω—à–µ n –ø–æ–≤—Ç–æ—Ä–µ–Ω—å

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –æ–∑–Ω–∞—á–∞—î, —â–æ —Å–∏–º–≤–æ–ª –º–∞—î –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è –ø—Ä–∏–Ω–∞–π–º–Ω—ñ n —Ä–∞–∑—ñ–≤, –∞–ª–µ –º–æ–∂–µ —ñ –±—ñ–ª—å—à–µ.

–ü—Ä–∏–∫–ª–∞–¥:
```python
import re

# –®—É–∫–∞—î –ø—Ä–∏–Ω–∞–π–º–Ω—ñ —Ç—Ä–∏ —Å–∏–º–≤–æ–ª–∏ 'a'
result = re.search(r'a{3,}', 'aa')
print(result)  # –í–∏–≤–µ–¥–µ: None, –±–æ —î —Ç—ñ–ª—å–∫–∏ 2 'a'

result = re.search(r'a{3,}', 'aaa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aaa', –±–æ —î —Ä—ñ–≤–Ω–æ 3 'a'

result = re.search(r'a{3,}', 'aaaa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aaaa', –±–æ —î –±—ñ–ª—å—à–µ 3 'a'
```

a{3,} –æ–∑–Ω–∞—á–∞—î, —â–æ –º–∞—î –±—É—Ç–∏ –ø—Ä–∏–Ω–∞–π–º–Ω—ñ 3 —Å–∏–º–≤–æ–ª–∏ a, –∞–ª–µ –¥–æ–ø—É—Å–∫–∞—î—Ç—å—Å—è –±—ñ–ª—å—à–µ.

`{n,m}` ‚Äî –≤—ñ–¥ n –¥–æ m –ø–æ–≤—Ç–æ—Ä–µ–Ω—å

–¶–µ–π –∫–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä –≤–∏–∑–Ω–∞—á–∞—î –¥—ñ–∞–ø–∞–∑–æ–Ω –∫—ñ–ª—å–∫–æ—Å—Ç—ñ –ø–æ–≤—Ç–æ—Ä–µ–Ω—å —Å–∏–º–≤–æ–ª—É ‚Äî –º—ñ–Ω—ñ–º—É–º n —ñ –º–∞–∫—Å–∏–º—É–º m.

–ü—Ä–∏–∫–ª–∞–¥:
```python

# –®—É–∫–∞—î –≤—ñ–¥ 2 –¥–æ 4 —Å–∏–º–≤–æ–ª—ñ–≤ 'a'
result = re.search(r'a{2,4}', 'a')
print(result)  # –í–∏–≤–µ–¥–µ: None, –±–æ —î —Ç—ñ–ª—å–∫–∏ 1 'a'

result = re.search(r'a{2,4}', 'aa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aa', –±–æ —î 2 'a'

result = re.search(r'a{2,4}', 'aaaaa')
print(result.group())  # –í–∏–≤–µ–¥–µ: 'aaaa', –±–æ –æ–±–º–µ–∂–µ–Ω–Ω—è –¥–æ 4 —Å–∏–º–≤–æ–ª—ñ–≤
```

a{2,4} –æ–∑–Ω–∞—á–∞—î, —â–æ –º–∞—î –±—É—Ç–∏ –≤—ñ–¥ 2 –¥–æ 4 —Å–∏–º–≤–æ–ª—ñ–≤ a. –Ø–∫—â–æ –≤ —Ä—è–¥–∫—É –±—ñ–ª—å—à–µ 4 —Å–∏–º–≤–æ–ª—ñ–≤, –±—É–¥–µ –∑–Ω–∞–π–¥–µ–Ω–æ —Ç—ñ–ª—å–∫–∏ –ø–µ—Ä—à—ñ 4.

üîö –ü—ñ–¥—Å—É–º–æ–∫:

`*` ‚Äî 0 –∞–±–æ –±—ñ–ª—å—à–µ –ø–æ–≤—Ç–æ—Ä–µ–Ω—å.

`+` ‚Äî 1 –∞–±–æ –±—ñ–ª—å—à–µ –ø–æ–≤—Ç–æ—Ä–µ–Ω—å.

`?` ‚Äî 0 –∞–±–æ 1 –ø–æ–≤—Ç–æ—Ä–µ–Ω–Ω—è.

`{n}` ‚Äî —Ä—ñ–≤–Ω–æ n –ø–æ–≤—Ç–æ—Ä–µ–Ω—å.

`{n,}` ‚Äî –Ω–µ –º–µ–Ω—à–µ n –ø–æ–≤—Ç–æ—Ä–µ–Ω—å.

`{n,m}` ‚Äî –≤—ñ–¥ n –¥–æ m –ø–æ–≤—Ç–æ—Ä–µ–Ω—å.

–ö–≤–∞–Ω—Ç–∏—Ñ—ñ–∫–∞—Ç–æ—Ä–∏ –¥–æ–ø–æ–º–∞–≥–∞—é—Ç—å —É—Ç–æ—á–Ω–∏—Ç–∏, —Å–∫—ñ–ª—å–∫–∏ —Ä–∞–∑—ñ–≤ –ø–æ–≤–∏–Ω–µ–Ω –∑—É—Å—Ç—Ä—ñ—á–∞—Ç–∏—Å—è –ø–µ–≤–Ω–∏–π –µ–ª–µ–º–µ–Ω—Ç —É —Ä—è–¥–∫—É, —â–æ–± –π–æ–≥–æ –º–æ–∂–Ω–∞ –±—É–ª–æ –≤–≤–∞–∂–∞—Ç–∏ –∑–±—ñ–≥–æ–º —ñ–∑ —à–∞–±–ª–æ–Ω–æ–º —Ä–µ–≥—É–ª—è—Ä–Ω–æ–≥–æ –≤–∏—Ä–∞–∑—É.

## –ì—Ä—É–ø—É–≤–∞–Ω–Ω—è —Ç–∞ –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–∏:
`( )` ‚Äî –≥—Ä—É–ø—É–≤–∞–Ω–Ω—è.


`|` ‚Äî –∞–±–æ (–∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–∞).
